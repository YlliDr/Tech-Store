### Component Usage Cheat Sheet

```html
<!-- Header -->
<script src="components/Header.js" defer></script>
<app-header></app-header>

<!-- Footer -->
<script src="components/Footer.js" defer></script>
<app-footer></app-footer>

<!-- Product Card -->
<script src="components/ProductCard.js" defer></script>
<product-card name="Item" price="99" image="img.jpg" 
              rating="5" reviews="10" link="product.html">
</product-card>

<!-- Review Card -->
<script src="components/ReviewCard.js" defer></script>
<review-card rating="5" title="Great!" author="John"
             date="01.01.2025" content="Excellent!">
</review-card>

<!-- Image Gallery -->
<script src="components/ImageGallery.js" defer></script>
<image-gallery images="img1.jpg,img2.jpg,img3.jpg">
</image-gallery>

<!-- Notification -->
<script src="components/Notification.js" defer></script>
<script>
  showToast('Message', 'success'); // or 'error', 'warning', 'info'
</script>
```

### Component Creation Cheat Sheet

```javascript
// 1. Create file: components/MyComponent.js

// 2. Define class
class MyComponent extends HTMLElement {
  connectedCallback() {
    // Get props
    const prop = this.getAttribute('prop') || 'default';
    
    // Render
    this.innerHTML = `<div>${prop}</div>`;
    
    // Add listeners
    this.querySelector('div').onclick = () => alert('Clicked!');
  }
}

// 3. Register
customElements.define('my-component', MyComponent);
```

```html
<!-- 4. Use in HTML -->
<script src="components/MyComponent.js" defer></script>
<my-component prop="value"></my-component>
```

**Full tutorial:** [COMPONENTS-GUIDE.md - Creating Components](COMPONENTS-GUIDE.md#-creating-your-own-component)
